    const int Createdir(const char* File)
    {
	union REGS InReg;
	union REGS OutReg;
	struct SREGS SegReg;

	InReg.h.ah=0x39;
	InReg.x.dx=FP_OFF(File);
	SegReg.ds=FP_SEG(File);

	intdosx(&InReg,&OutReg,&SegReg);
	return OutReg.x.cflag;

     }
